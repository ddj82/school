public String deleteFile(int r_no) {
		
		String fm = null;
		PreparedStatement ps = null;
		ResultSet rs = null;
		
		try {
			ps = con.prepareStatement("SELECT R_FILE FROM MC_ROOMS WHERE R_NO = ?");
			ps.setInt(1, r_no);
			rs = ps.executeQuery();
			
			if(rs.next())
				fm = rs.getString("R_FILE");
			
		} catch(Exception e) {
			System.out.println("파일 삭제 에러 : " + e);
		} finally {
			close(rs);
			close(ps);
		}
		
		return fm;
	}
